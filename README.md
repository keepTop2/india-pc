##### 目录结构

```
├── README.md
├── index.html
├── package-lock.json
├── package.json
├── public
│   ├── config.js   //接口地址
│   └── favicon.ico
├── src
│   ├── App.vue
│   ├── api         //后端接口调用
│   ├── components  //全局注册组件
│   ├── directive   //自定义指令
│   ├── hooks       //公共hooks
│   ├── i18n        //多语言
│   ├── layout      //布局
│   ├── main.ts
│   ├── router      //路由
│   ├── stores      //全局状态管理器
│   ├── styles      //全局样式
│   ├── types
│   ├── utils       //各模块公共方法
│   └── views       //对应路由下页面
├── tree.md
├── tsconfig.json
└── vite.config.ts
```
